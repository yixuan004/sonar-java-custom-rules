VariableLengthCheck Check The length of Member Variable should < 20

    public int abcdefghijklmnopqrst;  // Noncompliant
    public String abcdefghijklmnopqrs;


    public void tsrqponmlkjihgfedcba(String a) {
        System.out.println("Hello, World");
    }

    public int abc;
    public String abcdefghijklmnopqrs;


    public void tsrqponmlkjihgfedcba(String a) {
